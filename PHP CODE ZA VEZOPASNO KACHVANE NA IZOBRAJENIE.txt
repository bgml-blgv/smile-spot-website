<?php
// Директория за качване (не в public директория)
$uploadDir = __DIR__ . '/uploads/';
if (!is_dir($uploadDir)) {
    mkdir($uploadDir, 0755, true);
}

// Проверка дали има файл
if (isset($_FILES['photo']) && $_FILES['photo']['error'] === UPLOAD_ERR_OK) {
    
    $fileTmpPath = $_FILES['photo']['tmp_name'];
    $fileName    = $_FILES['photo']['name'];
    $fileSize    = $_FILES['photo']['size'];
    $fileType    = $_FILES['photo']['type'];

    // Позволени MIME типове
    $allowedMimeTypes = ['image/jpeg', 'image/png'];

    if (!in_array($fileType, $allowedMimeTypes)) {
        die("Грешка: Позволени са само JPG и PNG файлове.");
    }

    // Ограничаване на размера (пример 5MB)
    if ($fileSize > 5 * 1024 * 1024) {
        die("Грешка: Файлът е твърде голям.");
    }

    // Генериране на уникално име, за да не презаписва и да не се изпълнява
    $fileExtension = pathinfo($fileName, PATHINFO_EXTENSION);
    $newFileName = uniqid('img_', true) . '.' . $fileExtension;

    $destPath = $uploadDir . $newFileName;

    if (move_uploaded_file($fileTmpPath, $destPath)) {
        echo "Файлът е качен успешно: " . htmlspecialchars($newFileName);
    } else {
        echo "Грешка при качването на файла.";
    }
} else {
    echo "Няма избран файл или възникна грешка.";
}
